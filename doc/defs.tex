\newcommand{\easyfig}[4]{
\begin{figure}
\includegraphics[#2]{#1}
\caption{ \figlabel{#3} #4}
\end{figure}}

\newcommand{\includedot}[2]{

}

\newcommand{\dotfig}[4]{

\end{figure}}
